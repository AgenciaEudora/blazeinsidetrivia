Cómo conectar Google Sheets

1. Abre el spreadsheet de Blaze Inside (ya tiene una pestaña por socia).
2. Ve a Extensiones → Apps Script.
3. Borra el contenido y pega este código:

function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var d = JSON.parse(e.postData.contents); var nameLower = (d.playerName || '').toLowerCase().trim(); // Buscar hoja por nombre (exacto → parcial → fallback) var target = null; var sheets = ss.getSheets(); for (var i = 0; i < sheets.length; i++) { if (sheets[i].getName().toLowerCase().trim() === nameLower) { target = sheets[i]; break; } } if (!target) { for (var i = 0; i < sheets.length; i++) { var sn = sheets[i].getName().toLowerCase(); if (sn.indexOf(nameLower) !== -1 || nameLower.indexOf(sn) !== -1) { target = sheets[i]; break; } } } if (!target) { target = ss.getSheetByName('Respuestas') || ss.insertSheet('Respuestas'); } if (target.getLastRow() === 0) { target.appendRow([ 'Session ID','Timestamp','Jugadora', 'Post N°','Título','Categoría','Formato', '¿Qué llamó la atención?', '¿Lo publicarías?', '¿Qué mejorarías/agregarías?', 'Nota' ]); } target.appendRow([ d.sessionId, d.timestamp, d.playerName, d.postNumber, d.postTitulo, d.categoria, d.formato, d.llamoAtencion, d.loPublicarias, d.mejorariaAgregaria, d.nota || '' ]); return ContentService .createTextOutput(JSON.stringify({status:'ok',sheet:target.getName()})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error',msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput('Blaze Inside Posts Review API OK'); }
